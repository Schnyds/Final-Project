    <script>
        const questions = [
            // Era 1: Ancient Egypt
            { era: "Ancient Egypt", question: "Who was the first female pharaoh of Egypt?", choices: ["Cleopatra", "Nefertiti", "Hatshepsut", "Nefertari"], correct: 2 },
            { era: "Ancient Egypt", question: "What structure is the Sphinx associated with?", choices: ["Pyramids", "Colosseum", "Parthenon", "Stonehenge"], correct: 0 },
            
            // Add more eras with two questions each...
        ];

        let currentQuestionIndex = 0;
        let questionsAnsweredCorrectly = 0;
        let score = 0;
        let powerLevel = 100;

        function loadQuestion() {
            const questionData = questions[currentQuestionIndex];
            document.getElementById("era-display").textContent = questionData.era;
            document.getElementById("question").textContent = questionData.question;
            document.getElementById("choices").innerHTML = "";

            questionData.choices.forEach((choice, index) => {
                const button = document.createElement("button");
                button.textContent = choice;
                button.onclick = () => checkAnswer(index);
                document.getElementById("choices").appendChild(button);
            });
        }

        function checkAnswer(selected) {
            const questionData = questions[currentQuestionIndex];
            if (selected === questionData.correct) {
                score += 10;
                questionsAnsweredCorrectly += 1;
                document.getElementById("feedback").textContent = "Correct! Answer another question to move forward.";
                
                if (questionsAnsweredCorrectly < 2) {
                    currentQuestionIndex += 1; // Move to next question within the same era
                    loadQuestion();
                } else {
                    document.getElementById("next").style.display = "block"; // Show 'next' button to proceed to the next era
                }
            } else {
                score -= 5;
                powerLevel -= 10;
                document.getElementById("feedback").textContent = "Incorrect! You've been transported to another era.";
                currentQuestionIndex = Math.floor(Math.random() * questions.length); // Randomize era on incorrect answer
                questionsAnsweredCorrectly = 0; // Reset correct answers count if answer is incorrect
                loadQuestion();
            }

            document.getElementById("power-level").style.width = powerLevel + "%";
            document.getElementById("score").textContent = `Score: ${score}`;
            checkGameOver();
        }

        function nextEra() {
            currentQuestionIndex += 1; // Move to the first question of the next era
            questionsAnsweredCorrectly = 0; // Reset correct answers counter
            document.getElementById("next").style.display = "none"; // Hide the 'next' button
            document.getElementById("feedback").textContent = "";
            loadQuestion();
        }

        function checkGameOver() {
            if (powerLevel <= 0) {
                document.getElementById("question").textContent = "Game Over! Your power supply is depleted.";
                document.getElementById("choices").innerHTML = "";
                document.getElementById("next").style.display = "none";
            } else if (currentQuestionIndex >= questions.length) {
                showRank();
            }
        }

        function showRank() {
            let rank;
            if (score >= 100) rank = "Time Guardian";
            else if (score >= 80) rank = "Master Historian";
            else if (score >= 60) rank = "Chrono Explorer";
            else rank = "Apprentice Time Traveler";
            
            document.getElementById("question").textContent = `Congratulations! You've completed the game as a ${rank}.`;
            document.getElementById("choices").innerHTML = "";
            document.getElementById("next").style.display = "none";
        }

        loadQuestion();
    </script>
